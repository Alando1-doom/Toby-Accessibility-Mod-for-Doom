Actor RedCard_TO : CustomInventory 13
{
 inventory.pickupmessage "You got the red key card."
 inventory.pickupsound "misc/keycard"
 -COUNTITEM
 States
 {
  Spawn: 
	   RKEY A 0 A_SpawnItemEx("KeyPickupChecker", 0, 0, 0, 0, 0, 0, 0, SXF_NOCHECKPOSITION, 0)
       RKEY A 10
	   RKEY B 10 Bright
	   RKEY A 10
	   RKEY B 10 Bright
       Loop
 Pickup:
	   TNT1 A 0 A_PlaySound("ann/redcard")
       TNT1 A 1 A_GiveInventory("RedCard", 1)
	   Stop
    }
}

    	MONSTER 
        +JUSTHIT
        +DONTGIB
    	+FLOORCLIP
        +NOTARGET
        -COUNTKILL
        -SHOOTABLE
        +NOBLOOD
        +FASTER
        +ALWAYSFAST
		+NOCLIP
		+INVULNERABLE
		+GHOST
		+LOOKALLAROUND
		+SEEINVISIBLE
    	//PainSound "null"
        //DeathSound "null"
    	MeleeDamage 0
    	States
    	{
	Spawn: 
        TNT1 A 1 A_Look
		TNT1 A 0 A_CheckSight("Death")
        Goto See 
    See: 
        TNT1 A 1 A_Chase
		TNT1 A 0 A_CheckSight("Death")
		TNT1 A 1 A_CheckSight("Death")
        Goto ISeeYou
    Missile:
        TNT1 A 1 A_FaceTarget
		TNT1 A 0 A_CheckSight("Death")
		TNT1 A 1 A_CheckSight("Death")
        Goto ISeeYou
    Melee:
        TNT1 A 1 A_FaceTarget
		TNT1 A 0 A_CheckSight("Death")
		TNT1 A 1 A_CheckSight("Death")
        Goto ISeeYou
    Pain: 
        TNT1 A 1
        TNT1 A 1 A_Pain 
        Goto See
    ISeeYou:
         TNT1 A 1 A_PlaySoundEx("misc/key", "SoundSlot6", 0, 0)
		 TNT1 A 4
         Goto Death
    Death: 
        TNT1 A 1
        Stop
        }
}
